The C implementation (smollm2.c) suffers from:
- **7.1 tok/s** performance (vs100 tok/s target) due to unvectorized operations

Zig's @Vector provides:

- **Compile-time optimization** via comptime CPU feature detection
- **Type-safe memory** preventing buffer issues

### 4.1 Zig @Vector Primitive Usage for Matmul

#### Basic Vector Creation
```zig
// Simple vector of4 f32 elements
const a: @Vector(4, f32) = .{ 1.0, 2.0, 3.0, 4.0 };
const b: @Vector(4, f32) = .{ 5.0, 6.0, 7.0, 8.0 };

// Element-wise operations (auto-vectorized)
const c = a + b;  // .{6.0, 8.0, 10.0, 12.0 }
const d = a * b;  // .{ 5.0, 12.0, 21.0, 32.0 }
```

#### Dot Product via @reduce
```zig
pub inline fn dotProduct(comptime len: usize) f32 {
    const a: @Vector(len, f32) = ...;
    const b: @Vector(len, f32) = ...;
    // Multiply element-wise, then sum all lanes
    return @reduce(.Add, a * b);
}
```

#### Dynamic Vector Length Based on CPU
```zig
const std = @import("std");
const builtin = @import("builtin");

// Get optimal vector length at compile time
pub const VECTOR_LEN: usize = blk: {
    const cpu = std.Target.Cpu.baseline(
        std.Target.Cpu.Arch.x86_64,
        builtin.os
    );
    break :blk std.simd.suggestVectorLengthForCpu(f32, cpu);
};

// For ARM NEON: typically 4 f32 elements
// For AVX2: 8 f32 elements
// For AVX-512: 16 f32 elements
```

#### Matmul Kernel Pattern
```zig
// Inner loop kernel for matrix multiplication
pub fn matmulKernel(
    a: [*]const f32,
    b: [*]const f32,
    c: [*]f32,
    i: usize,
    j: usize,
    k_start: usize,
    k_end: usize,
    stride_a: usize,
    stride_b: usize,
 comptime VLEN: usize,
) void {
    var sum: f32 = 0;
    var k = k_start;

    while (k < k_end) : (k += VLEN) {
        // Load VLEN elements from A row i
        const va = @as(@Vector(VLEN, f32), a[i * stride_a + k ..][0..VLEN].*);
        // Load VLEN elements from B column j
        const vb = @as(@Vector(VLEN, f32), b[k * stride_b + j ..][0..VLEN].*);
        // Multiply and accumulate
        sum += @reduce(.Add, va * vb);
    }

    // Handle remaining elements (tail)
    while (k < k_end) : (k += 1) {
        sum += a[i * stride_a + k] * b[k * stride_b + j];
    }

    c[i * stride_b + j] = sum;
}
```

### 4.3 Memory Layout for Optimal Cache Usage

#### Cache-Blocking (Tiling) Pattern
```zig
const std = @import("std");

// Block size tuned for L1 cache (typically 32KB)
// For f32: 32 * 32 * 4 bytes = 4KB fits easily
const BLOCK_SIZE = 32;

pub fn blockedMatmul(
    a: []const f32,
    b: []const f32,
    c: []f32,
    n: usize,
) void {
    var i0: usize = 0;
    while (i0 < n) : (i0 += BLOCK_SIZE) {
        var j0: usize = 0;
        while (j0 < n) : (j0 += BLOCK_SIZE) {
            var k0: usize = 0;
            while (k0 < n) : (k0 += BLOCK_SIZE) {
                // Multiply i0:block x j0:block block with k0:block
                blockMultiply(a, b, c, i0, j0, k0, n);
            }
        }
    }
}

fn blockMultiply(
    a: []const f32,
    b: []const f32,
    c: []f32,
    i0: usize,
    j0: usize,
    k0: usize,
    n: usize,
) void {
    const i_end = @min(i0 + BLOCK_SIZE, n);
    const j_end = @min(j0 + BLOCK_SIZE, n);
    const k_end = @min(k0 + BLOCK_SIZE, n);

    var i = i0;
    while (i < i_end) : (i += 1) {
        var j = j0;
        while (j < j_end) : (j += 1) {
            var sum: f32 = 0;
            var k = k0;
            while (k < k_end) : (k += 1) {
                sum += a[i * n + k] * b[k * n + j];
            }
            c[i * n + j] += sum;
        }
    }
}
```

#### Row-Major vs Column-Major
```zig
// For LLM inference, row-major is typically better
// because we often do: y = x @ W where x is batched

// Row-major access pattern (cache-friendly for x row)
pub fn rowMajorDot(
    row: []const f32,      // x[i, :]
    matrix: []const f32,   // W[:, :]
    output: []f32,
    row_len: usize,
    mat_cols: usize,
    mat_rows: usize,
) void {
    for (0..mat_cols) |j| {
        var sum: f32 = 0;
        for (0..mat_rows) |k| {
            sum += row[k] * matrix[k * mat_cols + j];
        }
        output[j] = sum;
    }
}
```

### 4.4 Techniques for Specific Operations

#### 4.4.1 Dot Product Computation

```zig
// Basic vectorized dot product
pub fn dotProductVec(
    a: []const f32,
    b: []const f32,
    comptime VLEN: usize,
) f32 {
    var sum: f32 = 0;
    var i: usize = 0;

    // Vectorized loop
    while (i + VLEN <= a.len) : (i += VLEN) {
        const va = @as(@Vector(VLEN, f32), a[i..][0..VLEN].*);
        const vb = @as(@Vector(VLEN, f32), b[i..][0..VLEN].*);
        sum += @reduce(.Add, va * vb);
    }

    // Tail处理
    while (i < a.len) : (i += 1) {
        sum += a[i] * b[i];
    }

    return sum;
}

// Multi-register dot product for larger vectors
pub fn dotProductLarge(
    a: []const f32,
    b: []const f32,
) f32 {
    var sum0: f32 = 0;
    var sum1: f32 = 0;
    var sum2: f32 = 0;
    var sum3: f32 = 0;

    var i: usize = 0;
    while (i + 16 <= a.len) : (i += 16) {
        const va = @as(@Vector(4, f32), a[i..][0..4].*);
        const vb = @as(@Vector(4, f32), b[i..][0..4].*);
        sum0 += @reduce(.Add, va * vb);
 // ... repeat for next4-element chunks
    }

    return sum0 + sum1 + sum2 + sum3;
}
```

#### 4.4.2 Softmax

```zig
// Numerically stable softmax implementation
pub fn softmax(
    logits: []f32,
    output: []f32,
) void {
    // Step 1: Find max for numerical stability
    var max_val: f32 = -std.math.inf(f32);
    for (logits) |x| {
        if (x > max_val) max_val = x;
    }

    // Step 2: Compute exp(x - max) and sum
    var sum_exp: f32 = 0;
    for (logits, 0..) |x, i| {
        output[i] = @exp(x - max_val);
        sum_exp += output[i];
    }

    // Step 3: Normalize
    const inv_sum = 1.0 / sum_exp;
    for (output) |*x| {
        x.* *= inv_sum;
    }
}

// Vectorized softmax (approximate, for inference)
pub fn softmaxVec(
    logits: []const f32,
    output: []f32,
    comptime VLEN: usize,
) void {
    // Find max
    var max_buf: [VLEN]f32 = undefined;
    var global_max: f32 = -std.math.inf(f32);

    var i: usize = 0;
    while (i + VLEN <= logits.len) : (i += VLEN) {
        const vec = @as(@Vector(VLEN, f32), logits[i..][0..VLEN].*);
        const local_max = @reduce(.Max, vec);
        max_buf[0] = local_max;
        for (max_buf[1..VLEN], 1..) |*m, j| {
            m.* = max_buf[j - 1];
        }
        for (max_buf) |m| {
            if (m > global_max) global_max = m;
        }
    }

    // Handle tail
    for (logits[i..]) |x| {
        if (x > global_max) global_max = x;
    }

    // Compute exp and sum
    var sum_exp: f32 = 0;
    i = 0;
    while (i + VLEN <= logits.len) : (i += VLEN) {
        const vec = @as(@Vector(VLEN, f32), logits[i..][0..VLEN].*);
        const shifted = vec - @as(@Vector(VLEN, f32), @splat(global_max));
        const exp_vec = @exp(shifted);
        @memcpy(@as([*]f32, @ptrCast(&output[i])), @as([*]const f32, @ptrCast(&exp_vec)), VLEN);
        sum_exp += @reduce(.Add, exp_vec);
    }

    // Handle tail
    for (logits[i..], 0..) |x, j| {
        output[i + j] = @exp(x - global_max);
        sum_exp += output[i + j];
    }

    // Normalize
    const inv_sum = 1.0 / sum_exp;
    for (output) |*x| {
        x.* *= inv_sum;
    }
}
```

#### 4.4.3 Quantization/Dequantization

```zig
// INT8 quantization with scale factor
pub const Quantized = struct {
    data: []i8,
    scale: f32,
    zero_point: i32,
};

// Quantize f32 to i8
pub fn quantize(values: []const f32, output: []i8, scale: *f32) void {
    var min_val: f32 = std.math.inf(f32);
    var max_val: f32 = -std.math.inf(f32);

    for (values) |v| {
        if (v < min_val) min_val = v;
        if (v > max_val) max_val = v;
    }

    const range = max_val - min_val;
    scale.* = range / 255.0;

    for (values, 0..) |v, i| {
        const normalized = (v - min_val) / range;
        output[i] = @intFromFloat(normalized * 255.0);
    }
}

// Dequantize i8 back to f32
pub fn dequantize(
    quantized: []const i8,
    scale: f32,
    output: []f32,
) void {
    for (quantized, 0..) |q, i| {
        output[i] = @as(f32, @floatFromInt(q)) * scale;
    }
}

// Vectorized dequantization
pub fn dequantizeVec(
    quantized: []const i8,
    scale: f32,
    output: []f32,
    comptime VLEN: usize,
) void {
    const scale_vec = @as(@Vector(VLEN, f32), @splat(scale));

    var i: usize = 0;
    while (i + VLEN <= quantized.len) : (i += VLEN) {
        const qvec = @as(@Vector(VLEN, i8), quantized[i..][0..VLEN].*);
        const fvec = @as(@Vector(VLEN, f32), @bitCast(qvec));
        const out_vec = fvec * scale_vec;
        @memcpy(@as([*]f32, @ptrCast(&output[i])), @as([*]const f32, @ptrCast(&out_vec)), VLEN);
    }

    // Tail
    for (quantized[i..], 0..) |q, j| {
        output[i + j] = @as(f32, @floatFromInt(q)) * scale;
    }
}

// INT4 quantization (packed 2 per byte)
pub fn quantizeInt4(values: []const f32, output: []u8, scale: *f32) void {
    // Similar to INT8 but with 0-15 range
    var min_val: f32 = std.math.inf(f32);
    var max_val: f32 = -std.math.inf(f32);

    for (values) |v| {
        if (v < min_val) min_val = v;
        if (v > max_val) max_val = v;
    }

    const range = max_val - min_val;
    scale.* = range / 15.0;

    for (values, 0..) |v, i| {
        const normalized = (v - min_val) / range;
        output[i / 2] |= @intCast(normalized * 15.0) << (4 * (i % 2));
    }
}
```

### 4.5 Threading Patterns for Parallel Inference

#### Work-Pool Pattern with ThreadPool
```zig
const std = @import("std");
const ThreadPool = std.ThreadPool;
const Task = ThreadPool.Task;

pub fn parallelMatmul(
    a: []const f32,
    b: []const f32,
    c: []f32,
    n: usize,
    num_threads: usize,
) void {
    const pool = ThreadPool.init(.{
        .n_jobs = num_threads,
    });
    defer pool.deinit();

    const rows_per_thread = (n + num_threads - 1) / num_threads;

    var futures: [num_threads]Future = undefined;

    for (futures, 0..) |*f, t| {
        const start_row = t * rows_per_thread;
        const end_row = @min(start_row + rows_per_thread, n);

        f.* = pool.spawn(.{ .func = matmulRows, .args = .{
            a, b, c, start_row, end_row, n
        }});
    }

    for futures) |f| {
        f.wait();
    }
}

fn matmulRows(
    a: []const f32,
    b: []const f32,
    c: []f32,
    start_row: usize,
    end_row: usize,
    n: usize,
) void {
    for (start_row..end_row) |i| {
        for (0..n) |j| {
            var sum: f32 = 0;
            for (0..n) |k| {
                sum += a[i * n + k] * b[k * n + j];
            }
            c[i * n + j] = sum;
        }
    }
}
```

#### Chunk-Based Parallelism for Attention
```zig
pub fn parallelAttention(
    q: []const f32,
    k: []const f32,
    v: []const f32,
    output: []f32,
    num_heads: usize,
    seq_len: usize,
    head_dim: usize,
 num_threads: usize,
) void {
    const pool = ThreadPool.init(.{ .n_jobs = num_threads });
    defer pool.deinit();

    const heads_per_thread = (num_heads + num_threads - 1) / num_threads;

    for (0..num_threads) |t| {
        const start_head = t * heads_per_thread;
        const end_head = @min(start_head + heads_per_thread, num_heads);

        pool.spawn(.{ .func = attentionHeads, .args = .{
            q, k, v, output, start_head, end_head, seq_len, head_dim
        }});
    }
}

fn attentionHeads(
    q: []const f32,
    k: []const f32,
    v: []const f32,
    output: []f32,
    start_head: usize,
    end_head: usize,
    seq_len: usize,
    head_dim: usize,
) void {
    for (start_head..end_head) |h| {
        const q_offset = h * seq_len * head_dim;
        const k_offset = h * seq_len * head_dim;
        const v_offset = h * seq_len * head_dim;
        const out_offset = h * seq_len * head_dim;

        // Compute attention for this head
        computeAttention(
            q[q_offset..],
            k[k_offset..],
            v[v_offset..],
            output[out_offset..],
            seq_len,
            head_dim,
        );
    }
}
```

### 4.6 Arena Allocator Usage for Inference Buffers

#### Pre-Allocation Pattern
```zig
const std = @import("std");

pub const InferenceBuffer = struct {
    arena: std.heap.ArenaAllocator,
    logits: []f32,
    probs: []f32,
    temp_buffer: []f32,
    kv_cache: KVCache,

    pub fn init(allocator: std.mem.Allocator, config: ModelConfig) !@This() {
        var arena = std.heap.ArenaAllocator.init(allocator);
        errdefer arena.deinit();

        const seq_len = config.max_seq_len;
        const hidden = config.hidden_size;

        const logits = try arena.allocator().alloc(f32, hidden);
        const probs = try arena.allocator().alloc(f32, hidden);
        const temp = try arena.allocator().alloc(f32, seq_len * hidden);

        return @This(){
            .arena = arena,
            .logits = logits,
            .probs = probs,
            .temp_buffer = temp,
            .kv_cache = try KVCache.init(arena.allocator(), config),
        };
    }

    pub fn deinit(self: *@This()) void {
        self.kv_cache.deinit();
        self.arena.deinit();
    }

    pub fn reset(self: *@This()) void {
        // Reset state for new inference run
        @memset(self.logits, 0);
        @memset(self.probs, 0);
        @memset(self.temp_buffer, 0);
        self.kv_cache.reset();
    }
};

const KVCache = struct {
    key_cache: [][]f32,
    value_cache: [][]f32,

    pub fn init(allocator: std.mem.Allocator, config: ModelConfig) !@This() {
        const num_layers = config.num_layers;
        const num_heads = config.num_heads;
        const seq_len = config.max_seq_len;
        const head_dim = config.head_dim;

        var key_cache: [][]f32 = try allocator.alloc([]f32, num_layers * num_heads);
        var value_cache: [][]f32 = try allocator.alloc([]f32, num_layers * num_heads);

        for (key_cache, 0..) |*k, i| {
            k.* = try allocator.alloc(f32, seq_len * head_dim);
            @memset(k.*, 0);
        }

        for (value_cache, 0..) |*v, i| {
            v.* = try allocator.alloc(f32, seq_len * head_dim);
            @memset(v.*, 0);
        }

        return @This(){
            .key_cache = key_cache,
            .value_cache = value_cache,
        };
    }

    pub fn deinit(self: *@This()) void {
        // Arenas handle cleanup automatically
    }

    pub fn reset(self: *@This()) void {
        for (self.key_cache) |k| {
            @memset(k, 0);
        }
        for (self.value_cache) |v| {
            @memset(v, 0);
        }
    }
};
```

#### Thread-Local Arena for Parallel Workers
```zig
pub fn runInferenceParallel(
    model: *Model,
    input_ids: []const u32,
    output_ids: []u32,
    num_threads: usize,
 allocator: std.mem.Allocator,
) !void {
    // Create one arena per thread for thread safety
    var arenas: [num_threads]std.heap.ArenaAllocator = undefined;
    for (&arenas) |*arena| {
        arena.* = std.heap.ArenaAllocator.init(allocator);
    }
    defer {
        for (arenas) |*arena| {
            arena.deinit();
        }
    }

    const pool = ThreadPool.init(.{ .n_jobs = num_threads });
    defer pool.deinit();

    // Each layer processed by different thread
    for (0..model.config.num_layers) |layer_idx| {
        const thread_idx = layer_idx % num_threads;
        const thread_arena = &arenas[thread_idx];

        pool.spawn(.{ .func = processLayer, .args = .{
            model, layer_idx, input_ids, thread_arena
        }});
    }
}
```

## 5. Pseudocode for Core Inference Loop

```
function inference(input_ids):
    buffers = InferenceBuffer.init(allocator, config)
    position = 0

    while position < max_seq_len:
        // 1. Embed input
        embeddings = embedLayer.forward(input_ids[position])

        // 2. Process all layers (could be parallelized)
        for each layer in model.layers:
            // a. Attention with KV cache
            attn_output = attention(
                embeddings,
                layer.kv_cache.keys[position],
                layer.kv_cache.values[position]
            )

            // b. Feed-forward
            ffn_output = layer.ffn.forward(attn_output)

            // c. Residual connection
            embeddings = embeddings + ffn_output

        // 3. Final layer norm
        hidden = final_layer_norm(embeddings)

        // 4. Project to vocabulary
        logits = lm_head.forward(hidden)

        // 5. Sample next token
        next_token = sample(logits)

        // 6. Check for EOS
        if next_token == EOS_TOKEN:
            break

        output_ids[position] = next_token
        position += 1

 return output_ids
```

## 6. Performance Optimization Checklist

### SIMD Vectorization
- [ ] Use `@Vector` with power-of-2 lengths (4, 8, 16)
- [ ] Prefer `@reduce(.Add, a * b)` for dot products
- [ ] Use `@splat` for broadcast operations
- [ ] Handle tail elements with scalar fallback
- [ ] Use `@select` for conditional moves

### Memory Layout
- [ ] Use cache-blocking with BLOCK_SIZE = 32 (fits in L1)
- [ ] Align allocations to 32 or 64 bytes
- [ ] Prefer row-major for attention inputs
- [ ] Pre-allocate buffers to avoid reallocation

### Threading
- [ ] Use ThreadPool for embarrassingly parallel operations
- [ ] Partition work by rows/heads/layers
- [ ] Use thread-local arenas for thread safety
- [ ] Minimize synchronization points

### Numerical Stability
- [ ] Softmax: subtract max before exp
- [ ] LayerNorm: compute mean and variance carefully
- [ ] Use f32 for inference (f16 requires careful handling)
